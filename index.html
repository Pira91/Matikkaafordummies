function calculateTrig() { // Hae kulma asteina ja muunna radiaaneiksi const angleDeg = parseFloat(document.getElementById('trig-angle').value); const angleRad = angleDeg * (Math.PI / 180); // Laske trigonometriset funktiot const sinValue = Math.sin(angleRad).toFixed(4); const cosValue = Math.cos(angleRad).toFixed(4); let tanValue = Math.tan(angleRad).toFixed(4); // Korjaa tangentin ääriarvot if (angleDeg % 90 === 0 && angleDeg % 180 !== 0) { tanValue = "∞"; } // Päivitä tulokset document.getElementById('sin-result').textContent = sinValue; document.getElementById('cos-result').textContent = cosValue; document.getElementById('tan-result').textContent = tanValue; // Animaatio yksikköympyrälle updateUnitCircle(angleDeg); } function updateUnitCircle(angleDeg) { const angleRad = angleDeg * (Math.PI / 180); const centerX = 75; const centerY = 75; const radius = 70; // Laske pisteen sijainti ympyrällä const pointX = centerX + radius * Math.cos(angleRad); const pointY = centerY - radius * Math.sin(angleRad); // Y-akseli on käänteinen SVG:ssä // Päivitä viiva ja piste document.getElementById('trig-circle-line').setAttribute("d", `M${centerX} ${centerY} L${pointX} ${pointY}`); document.getElementById('trig-circle-point').setAttribute("cx", pointX); document.getElementById('trig-circle-point').setAttribute("cy", pointY); document.getElementById('trig-circle-point').style.opacity = "1"; } /* Lisää tähän osaan jos haluat muokata ulkoasua */ .calculator h3 { margin-top: 0; color: #3498db; font-size: 1.1em; } input[type="number"] { width: 100px; } svg { border: 1px solid #eee; border-radius: 50%; background-color: white; }
